CODIGO SEGURO

import sqlite3
import logging  # Cambiamos os.system por logging
from flask import Flask, request, abort

app = Flask(__name__)

# Configuración básica de logging
logging.basicConfig(level=logging.INFO)

@app.route('/search')
def search():
    user_query = request.args.get('id')
    
    # 1. Validación de entrada: Nos aseguramos de que el ID sea lo que esperamos
    if not user_query or not user_query.isalnum():
        abort(400, description="ID inválido")

    conn = sqlite3.connect('database.db')
    cursor = conn.cursor()

    # 2. SOLUCIÓN SQL INJECTION: Usar consultas parametrizadas (?)
    # Nunca uses f-strings o concatenación en SQL
    query = "SELECT * FROM users WHERE id = ?"
    cursor.execute(query, (user_query,))
    
    # 3. SOLUCIÓN COMMAND INJECTION: Usar librerías nativas de Python (logging)
    # Evita os.system() a toda costa para imprimir o procesar datos
    logging.info("Buscando al usuario %s", user_query)
    
    return f"Resultado para {user_query}"

if __name__ == "__main__":
    # 4. SOLUCIÓN CONFIGURACIÓN: Debug en False y host seguro
    # Bandit marcará error si ve debug=True o host='0.0.0.0'
    app.run(debug=False, host='127.0.0.1', port=5000)
